How to update?

All specific CSS has been placed in to the bootystrap-theme.Css file. Please download the files or use the CDN stylesheet for FuelUX and include the bootstrap-theme.css file

Getting started

  1. Add the fueluxclass to a page wrapper (usually either <html> tag or <body> tag)

	function customColumnRenderer(helpers, callback) {
	    // determine what column is being rendered
	    var column = helpers.columnAttr;

	    // get all the data for the entire row
	    var rowData = helpers.rowData;
	    var customMarkup = '';

	    // only override the output for specific columns.
	    // will default to output the text value of the row item
	    switch (column) {
	        case 'txtIcon':
	            // let's combine name and description into a single column
	            customMarkup = '<i class="fa fa-' + rowData.txtIcon + '"></i>';
	            break;
	        case 'boolSelected':
	            customMarkup = '<div class="checkbox"><label class="checkbox-custom ' + (rowData.boolSelected ? 'checked' : '') + '" data-initialize="checkbox"><input class="sr-only" type="checkbox" value="" ' + (rowData.boolSelected ? 'checked="checked"' : '') + '></label></div>';
	            break;
	        default:
	            // otherwise, just use the existing text value
	            customMarkup = helpers.item.text();
	            break;
	    }

	    helpers.item.html(customMarkup);

	    callback();
	}

	function customRowRenderer(helpers, callback) {
	    // let's get the id and add it to the "tr" DOM element
	    var item = helpers.item;
	    var disabled = (Math.random() < .25) ? "disabled" : "";
	    item.attr('class', disabled);

	    callback();
	}

	function staticDataSource(options, callback) {

	    // define the columns for the grid
	    var columns = [{
	        'label': '<input type="checkbox">', // column header label
	        'property': 'boolSelected', // the JSON property you are binding to
	        'sortable': false // is the column sortable?
	    }, {
	        'label': '',
	        'property': 'txtIcon',
	        'sortable': false
	    }, {
	        'label': 'Code',
	        'property': 'txtCode',
	        'sortable': true
	    }, {
	        'label': 'Activiteit',
	        'property': 'txtDescription',
	        'sortable': true
	    }];





	    // generate the rows in your dataset
	    // note: the property names of your items should

	    function generateRandomData() {
	        var items = [];
	        var icons = ['cutlery', 'car', 'camera', 'university'];
	        var code = ['WM', 'QSD', 'AZE'];
	        var activiteiten = ['Warme Maaltijd', 'Naschoolse opvang', 'Schooluitstap'];

	        function getRandom(arr) {
	            var min = 0;
	            var max = 2;
	            var index = Math.floor(Math.random() * (max - min + 1)) + min;
	            return arr[index];
	        }

	        for (var i = 1; i <= 50; i++) {
	            var item = {
	                id: i,
	                boolSelected: (Math.random() < .25),
	                txtDescription: getRandom(activiteiten),
	                txtCode: getRandom(code),
	                txtIcon: getRandom(icons)
	            }
	            items.push(item);
	        };
	        console.log(JSON.stringify(items));
	        return items;
	    }

	    items = generateRandomData();

	    // transform array
	    var pageIndex = options.pageIndex;
	    var pageSize = options.pageSize;
	    var totalItems = items.length;
	    var totalPages = Math.ceil(totalItems / pageSize);
	    var startIndex = (pageIndex * pageSize) + 1;
	    var endIndex = (startIndex + pageSize) - 1;
	    if (endIndex > items.length) {
	        endIndex = items.length;
	    }

	    var rows = items.slice(startIndex - 1, endIndex);

	    // define the datasource
	    var dataSource = {
	        'page': pageIndex,
	        'pages': totalPages,
	        'count': totalItems,
	        'start': startIndex,
	        'end': endIndex,
	        'columns': columns,
	        'items': rows
	    };

	    // pass the datasource back to the repeater
	    callback(dataSource);
	}

	            break;
	        case 'txtType':
	            customMarkup = '<select class="form-control"><option value="' + rowData.txtType + '" selected> ' + rowData.txtType + '</option><option value="Ja/Nee">Ja/Nee</option><option value="Tekst">Tekst</option><option>4</option> <option>5</option></select>';
